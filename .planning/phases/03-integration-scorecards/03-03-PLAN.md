```javascript
/* ── EMAIL CAPTURE WIRING (INTG-03) ── */
(function () {
  var emailForm = document.getElementById("n1-tool-email-form");
  var emailInput = document.getElementById("n1-tool-email");
  var successMsg = document.getElementById("n1-tool-email-success");
  if (!emailForm || !emailInput || !successMsg) return;

  emailForm.addEventListener("submit", function (e) {
    e.preventDefault();
    var email = emailInput.value.trim();
    if (!email) return;
    fetch(emailForm.action, {
      method: "POST",
      mode: "no-cors",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: "email=" + encodeURIComponent(email) + "&tag=tool-completion",
    })
      .then(function () {
        emailForm.style.display = "none";
        successMsg.style.display = "block";
      })
      .catch(function () {
        emailForm.submit();
      });
  });
})();
```

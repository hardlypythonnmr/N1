Do NOT duplicate the drawing code. Instead, extract the drawing logic into a private `drawScorecardToContext(ctx, config, width, height)` inner function, and have both `generateScorecard` and `generateScorecardPreview` call it. The private function signature:

```javascript
function drawScorecardToContext(ctx, config, canvasWidth, canvasHeight) {
  // All current drawing code from generateScorecard goes here
  // Replace 1200 with canvasWidth, 628 with canvasHeight throughout
  // proportional adjustments: header=Math.round(canvasHeight*0.127), footer y=Math.round(canvasHeight*0.892)
}
```

Then:

```javascript
N1.generateScorecard = function (config) {
  var canvas = document.createElement("canvas");
  canvas.width = 1200;
  canvas.height = 628;
  var ctx = canvas.getContext("2d");
  if (!ctx) {
    console.error("[N1] generateScorecard: Canvas 2D not available");
    return null;
  }
  drawScorecardToContext(ctx, config, 1200, 628);
  return canvas.toDataURL("image/png");
};

N1.generateScorecardPreview = function (config) {
  var canvas = document.createElement("canvas");
  canvas.width = 600;
  canvas.height = 314;
  var ctx = canvas.getContext("2d");
  if (!ctx) {
    console.error("[N1] generateScorecardPreview: Canvas 2D not available");
    return null;
  }
  ctx.scale(0.5, 0.5);
  drawScorecardToContext(ctx, config, 1200, 628);
  return canvas.toDataURL("image/png");
};
```

const GHOST_SIGNALS = [
{ id: 'no_salary_range', label: 'No indica rango salarial', weight: 15,
detect: (t) => !/(salario|sueldo|compensacion|\$|COP|USD|salary)/i.test(t) },
{ id: 'no_company_name', label: 'No menciona la empresa', weight: 20,
detect: (t) => /(empresa confidencial|confidential|cliente de)/i.test(t) },
{ id: 'vague_role', label: 'Descripcion del rol vaga', weight: 15,
detect: (t) => t.length < 400 },
{ id: 'unrealistic_requirements', label: 'Requisitos irreales para el nivel', weight: 10,
detect: (t) => { const m = t.match(/(\d+)\+?\s*a[no]s/gi)||[]; return m.reduce((s,x)=>s+parseInt(x),0)>8; }
},
{ id: 'urgent_language', label: 'Lenguaje de urgencia vacia', weight: 5,
detect: (t) => /(inmediatamente|urgente|ya mismo|asap)/i.test(t) },
{ id: 'generic_benefits', label: 'Beneficios genericos', weight: 5,
detect: (t) => /(excelente ambiente|beneficios competitivos)/i.test(t) && !/(seguro|prima|bono)/i.test(t) },
{ id: 'no_team_context', label: 'No describe equipo ni manager', weight: 10,
detect: (t) => !/(equipo|manager|reporta|team)/i.test(t) },
{ id: 'email_only', label: 'Solo acepta CV por email', weight: 5,
detect: (t) => /(enviar cv|manda tu hoja de vida)/i.test(t) && !/(aplica|apply|portal)/i.test(t) },
{ id: 'reposted', label: 'Indicios de reposteo multiple', weight: 10,
detect: (t) => /(actualiz|updated|repost|volvemos a publicar)/i.test(t) },
{ id: 'level_mismatch', label: 'Nivel no coincide con requisitos', weight: 5,
detect: (t) => /(junior|entry.level)/i.test(t) && /(\d+\+?\s*a[no]s|maestria|certificacion)/i.test(t) }
];

function analyzePosting(text) {
const signals = GHOST_SIGNALS.map(function(s) {
return Object.assign({}, s, { fired: s.detect(text) });
});
const score = signals.filter(function(s) { return s.fired; })
.reduce(function(sum, s) { return sum + s.weight; }, 0);
return { score: score, signals: signals };
}

function getVerdict(score) {
if (score <= 25) return { label: 'Probablemente real', color: '#22c55e' };
if (score <= 50) return { label: 'Oferta sospechosa', color: '#f59e0b' };
if (score <= 75) return { label: 'Alta probabilidad de ghost job', color: '#E85D26' };
return { label: 'Ghost job confirmado', color: '#dc2626' };
}

function getInterpretation(score) {
if (score <= 25) return 'Esta oferta muestra senales de ser legitima. Igual verificá la empresa en LinkedIn antes de aplicar.';
if (score <= 50) return 'Esta oferta tiene algunas senales de alerta. Busca la empresa en LinkedIn y verificá que el rol existe antes de invertir tiempo.';
if (score <= 75) return 'Esta oferta tiene muchas senales de ser un ghost job. Aplicar puede ser una perdida de tiempo y energia. Priorizá ofertas con menos senales de alerta.';
return 'Esta oferta tiene todas las senales de un ghost job. Muy probablemente no hay una posicion real detras. Usa ese tiempo en ofertas verificables.';
}
